// 5. A* Pathfinding Visualizer
const GRID_W = 40, GRID_H = 18;

function createGrid(wallDensity = 0.28) {
  return Array.from({ length: GRID_H }, (_, r) =>
    Array.from({ length: GRID_W }, (_, c) =>
      (r === 0 && c === 0) || (r === GRID_H-1 && c === GRID_W-1)
        ? 0 : (Math.random() < wallDensity ? 1 : 0)
    )
  );
}

function heuristic(a, b) {
  return Math.abs(a.r - b.r) + Math.abs(a.c - b.c);
}

function aStar(grid, start, end) {
  const key = (r, c) => `${r},${c}`;
  const open = [{ ...start, g: 0, f: heuristic(start, end), parent: null }];
  const closed = new Set();
  const best = { [key(start.r, start.c)]: 0 };

  while (open.length) {
    open.sort((a, b) => a.f - b.f);
    const cur = open.shift();
    const k = key(cur.r, cur.c);
    if (closed.has(k)) continue;
    closed.add(k);

    if (cur.r === end.r && cur.c === end.c) {
      const path = [];
      let node = cur;
      while (node) { path.unshift({ r: node.r, c: node.c }); node = node.parent; }
      return { path, visited: closed.size };
    }

    for (const [dr, dc] of [[-1,0],[1,0],[0,-1],[0,1]]) {
      const nr = cur.r + dr, nc = cur.c + dc;
      if (nr < 0 || nr >= GRID_H || nc < 0 || nc >= GRID_W) continue;
      if (grid[nr][nc] === 1) continue;
      const nk = key(nr, nc);
      if (closed.has(nk)) continue;
      const g = cur.g + 1;
      if (best[nk] !== undefined && best[nk] <= g) continue;
      best[nk] = g;
      open.push({ r: nr, c: nc, g, f: g + heuristic({ r: nr, c: nc }, end), parent: cur });
    }
  }
  return null;
}

function render(grid, path, visited) {
  const pathSet = new Set(path?.map(p => `${p.r},${p.c}`) ?? []);
  const rows = grid.map((row, r) =>
    row.map((cell, c) => {
      const k = `${r},${c}`;
      if (r === 0 && c === 0) return "S";
      if (r === GRID_H-1 && c === GRID_W-1) return "E";
      if (pathSet.has(k)) return "█";
      return cell === 1 ? "▓" : "·";
    }).join("")
  );
  return rows.join("\n");
}

class LinkedList:
    class Node:
        def __init__(self, element):
            self.element = element
            self.next = None

    def __init__(self):
        self.length = 0
        self.head = None

    def is_empty(self):
        return self.length == 0

    def add(self, element):
        node = self.Node(element)
        if self.is_empty():
            self.head = node
        else:
            current_node = self.head
            while current_node.next is not None:
                current_node = current_node.next
            current_node.next = node
        self.length += 1

    def remove(self, element):
        previous_node = None
        current_node = self.head
        while current_node is not None and current_node.element != element:
            previous_node = current_node
            current_node = current_node.next
        if current_node is None:
            return
        elif previous_node is not None:
            previous_node.next = current_node.next
        else:
            self.head = current_node.next
        self.length -= 1

        self.width = width

    def set_height(self, height):
        self.height = height

    def get_area(self):
        return self.width * self.height

    def get_perimeter(self):
        return 2 * (self.width + self.height)

    def get_diagonal(self):
        return math.sqrt(self.width**2 + self.height**2)

    def get_picture(self):
        if self.width > 50 or self.height > 50:
            return "Too big for picture."
        picture = ""
        for _ in range(self.height):
            picture += "*" * self.width + "\n"
        return picture
